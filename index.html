flyLabel.js

Flexible floating form labels to make your boring placeholders super fly. Based on the FloatLabel pattern by Matt Smith. With flyLabel.js, it's easy to trigger custom behaviors on form labels. Check out the examples, and contribute your own.

classic

.fly__label--classic {
  opacity: 0;
  top: 2em;
  z-index: -1;
}
.fly__label--classic.fly__label--active {
  opacity: 1;
  top: 1em;
  z-index: 1;
}

drop

.fly__label--drop {
  opacity: 0;
  text-shadow: 0 0 25px blue;
  -webkit-transform: scale(11);
  -moz-transform:    scale(11);
  -o-transform:      scale(11);
  transform:         scale(11);
  z-index: -1;
}
.fly__label--drop.fly__label--active {
  opacity: 1;
  text-shadow: 0 0 0px blue;
  -webkit-transform: scale(1);
  -moz-transform:    scale(1);
  -o-transform:      scale(1);
  transform:         scale(1);
  z-index: 1;
}

from-right

.fly__label--from-right {
  left: 5em;
  opacity: 0;
}
.fly__label--from-right.fly__label--active {
  left: 0%;
  opacity: 1;
}

spin

.fly__label--spin {
  opacity: 0;
}
.fly__label--spin > span {
  transition: all .5s ease;
  display: inline-block;
  -webkit-transform: rotateZ(180deg);
  -moz-transform:    rotateZ(180deg);
  -o-transform:      rotateZ(180deg);
  transform:         rotateZ(180deg);
}
.fly__label--spin.fly__label--active {
  opacity: 1;
}
.fly__label--spin.fly__label--active > span {
  -webkit-transform: rotateZ(0deg);
  -moz-transform:    rotateZ(0deg);
  -o-transform:      rotateZ(0deg);
  transform:         rotateZ(0deg);
}

upside-down

.fly__label--upside-down {
  opacity: 0;
}    
.fly__label--upside-down > span {
  transition: all .5s ease;
  position: relative;
}       
.fly__label--upside-down > span:nth-child(odd) {
  top: -5em;
}          
.fly__label--upside-down > span:nth-child(even) {
  top: 5em;
}          
.fly__label--upside-down > span:first-child {
  top: 0;
  left: -5em;
}           
.fly__label--upside-down > span:last-child {
  top: 0;
  right: -5em;
}           
.fly__label--upside-down.fly__label--active {
  opacity: 1;
}    
.fly__label--upside-down.fly__label--active > span {
  top: 0;
}    
.fly__label--upside-down.fly__label--active > span:first-child {
  left: 0;
}    
.fly__label--upside-down.fly__label--active > span:last-child {
  right: 0;
}
